const registerForm = document.getElementById('registerForm'); const messageDiv = document.getElementById('message'); registerForm.addEventListener('submit', async (event) => { event.preventDefault(); // Prevent the default form submission const email = document.getElementById('registerEmail').value; const password = document.getElementById('registerPassword').value; const role = document.getElementById('registerRole').value; const response = await fetch('https://yourwebsite.epizy.com/register.php', { method: 'POST', headers: { 'Content-Type': 'application/x-www-form-urlencoded', }, body: `email=${email}&password=${password}&role=${role}`, }); const result = await response.text(); messageDiv.innerHTML = result; // Display backend message });
